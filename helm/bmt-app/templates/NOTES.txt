Get the API URL by running these commands:
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app=bmt-api" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://127.0.0.1:8080 to access your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:3000
